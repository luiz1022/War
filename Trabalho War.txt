#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#define MAX_NOME 30
#define MAX_COR 10
#define NUM_TERRITORIOS 5

typedef struct {
    char nome[MAX_NOME];
    char cor[MAX_COR];
    int tropas;
} Territorio;

void limpar_quebra_linha(char *str) {
    size_t len = strlen(str);
    if (len > 0 && str[len - 1] == '\n') str[len - 1] = '\0';
}

int main() {
    Territorio mapa[NUM_TERRITORIOS];
    int i;

    printf("=== CADASTRO DE TERRITÓRIOS WAR ===\n");

    for (i = 0; i < NUM_TERRITORIOS; i++) {
        printf("\n-- Território %d --\n", i + 1);

        printf("Nome: ");
        fgets(mapa[i].nome, MAX_NOME, stdin);
        limpar_quebra_linha(mapa[i].nome);

        printf("Cor do Exército: ");
        fgets(mapa[i].cor, MAX_COR, stdin);
        limpar_quebra_linha(mapa[i].cor);

        printf("Tropas: ");
        if (scanf("%d", &mapa[i].tropas) != 1) return 1;
        while (getchar() != '\n');
    }

    printf("\n=== TERRITÓRIOS CADASTRADOS ===\n");
    for (i = 0; i < NUM_TERRITORIOS; i++) {
        printf("\n[%d] %s\nCor: %s\nTropas: %d\n", 
                i + 1, mapa[i].nome, mapa[i].cor, mapa[i].tropas);
    }

    printf("\nCadastro concluído.\n");
    return 0;
}
